Commerce Kickstart - text do kategórie

2 september, 2014 - 14:55 poslal eva_n

Zdravím všetkých,

snažím sa prevádzkovať svoj e-shop en4you.sk postavený na Commerce Kickstart (som začiatočníčka) a chcela by som zobraziť text ku každej kategórii produktov. Neviem si poradiť. Mám vyplnené pole „Popis“ pre každú kategóriu (termín taxonóme admin/structu­re/taxonomy/co­llection), skúšala som pridať pole do view, asi to nie je správna cesta. Zobrazilo sa niečo iné.

Mohli by ste mi prosím poradiť ako dostať text medzi nadpis kategórie a zoznam produktov v danej kategórii? Budem vďačná za radu.

  • Vytvoriť vlastné view, ktoré vypisuje len to description podľa argumentu
  • do view „Collection products“, ktoré vypisuje tú kategóriu sa to dá potom vložiť do headeru, ako view area (treba zaškrtnúť „Inherit contextual filters“)
  • a samozrejme pri kategóriách mať vypísané description

vlastné view s description podľa argumentu:

$view = new view();
$view->name = 'collection_description';
$view->description = '';
$view->tag = 'default';
$view->base_table = 'taxonomy_term_data';
$view->human_name = 'Collection - description';
$view->core = 7;
$view->api_version = '3.0';
$view->disabled = FALSE; /* Edit this to true to make a default view disabled initially */

/* Display: Master */
$handler = $view->new_display('default', 'Master', 'default');
$handler->display->display_options['use_more_always'] = FALSE;
$handler->display->display_options['use_more_text'] = 'viac';
$handler->display->display_options['access']['type'] = 'perm';
$handler->display->display_options['cache']['type'] = 'time';
$handler->display->display_options['cache']['results_lifespan'] = '21600';
$handler->display->display_options['cache']['results_lifespan_custom'] = '0';
$handler->display->display_options['cache']['output_lifespan'] = '21600';
$handler->display->display_options['cache']['output_lifespan_custom'] = '0';
$handler->display->display_options['query']['type'] = 'views_query';
$handler->display->display_options['exposed_form']['type'] = 'basic';
$handler->display->display_options['exposed_form']['options']['submit_button'] = 'Použiť';
$handler->display->display_options['exposed_form']['options']['exposed_sorts_label'] = 'Zoradiť podľa';
$handler->display->display_options['exposed_form']['options']['sort_asc_label'] = 'Vzostupne';
$handler->display->display_options['exposed_form']['options']['sort_desc_label'] = 'Zostupne';
$handler->display->display_options['pager']['type'] = 'none';
$handler->display->display_options['pager']['options']['offset'] = '0';
$handler->display->display_options['style_plugin'] = 'default';
$handler->display->display_options['style_options']['default_row_class'] = FALSE;
$handler->display->display_options['style_options']['row_class_special'] = FALSE;
$handler->display->display_options['row_plugin'] = 'fields';
$handler->display->display_options['row_options']['default_field_elements'] = FALSE;
/* Pole: Termín taxonómie: Popis výrazu */
$handler->display->display_options['fields']['description']['id'] = 'description';
$handler->display->display_options['fields']['description']['table'] = 'taxonomy_term_data';
$handler->display->display_options['fields']['description']['field'] = 'description';
$handler->display->display_options['fields']['description']['label'] = '';
$handler->display->display_options['fields']['description']['element_class'] = 'description';
$handler->display->display_options['fields']['description']['element_label_colon'] = FALSE;
$handler->display->display_options['fields']['description']['element_default_classes'] = FALSE;
/* Pole: Globálny: Custom text */
$handler->display->display_options['fields']['nothing']['id'] = 'nothing';
$handler->display->display_options['fields']['nothing']['table'] = 'views';
$handler->display->display_options['fields']['nothing']['field'] = 'nothing';
$handler->display->display_options['fields']['nothing']['label'] = '';
$handler->display->display_options['fields']['nothing']['alter']['text'] = '<div class="arrow"></div>';
$handler->display->display_options['fields']['nothing']['element_label_colon'] = FALSE;
$handler->display->display_options['fields']['nothing']['element_default_classes'] = FALSE;
/* Contextual filter: Termín taxonómie: ID termínu */
$handler->display->display_options['arguments']['tid']['id'] = 'tid';
$handler->display->display_options['arguments']['tid']['table'] = 'taxonomy_term_data';
$handler->display->display_options['arguments']['tid']['field'] = 'tid';
$handler->display->display_options['arguments']['tid']['default_action'] = 'default';
$handler->display->display_options['arguments']['tid']['exception']['title'] = 'Všetky';
$handler->display->display_options['arguments']['tid']['default_argument_type'] = 'taxonomy_tid';
$handler->display->display_options['arguments']['tid']['summary']['number_of_records'] = '0';
$handler->display->display_options['arguments']['tid']['summary']['format'] = 'default_summary';
$handler->display->display_options['arguments']['tid']['summary_options']['items_per_page'] = '25';
/* Filter criterion: Slovník taxonómie: Strojový názov */
$handler->display->display_options['filters']['machine_name']['id'] = 'machine_name';
$handler->display->display_options['filters']['machine_name']['table'] = 'taxonomy_vocabulary';
$handler->display->display_options['filters']['machine_name']['field'] = 'machine_name';
$handler->display->display_options['filters']['machine_name']['value'] = array(
  'collection' => 'collection',
);
$translatables['collection_description'] = array(
  t('Master'),
  t('viac'),
  t('Použiť'),
  t('Reset'),
  t('Zoradiť podľa'),
  t('Vzostupne'),
  t('Zostupne'),
  t('<div class="arrow"></div>'),
  t('Všetky'),
);
